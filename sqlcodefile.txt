//student table code 
CREATE TABLE STUDENTS (
    studentID NUMBER GENERATED ALWAYS AS IDENTITY (START WITH 100),
    fName VARCHAR2(64),
    lName VARCHAR2(64),
    email VARCHAR2(128) UNIQUE,
    "year" Varchar(64) CHECK("year" IN ('freshman', 'sophomore', 'junior', 'senior')),
    PRIMARY KEY (studentID)
);

//instructors table code

CREATE TABLE INSTRUCTORS (
    instructorID NUMBER GENERATED ALWAYS AS IDENTITY (START WITH 200),
    fName VARCHAR2(64),
    lName VARCHAR2(64),
    email VARCHAR2(64) UNIQUE,
    department VARCHAR2(64),
    PRIMARY KEY (instructorID)
);

//courses table code

CREATE TABLE COURSES (
    courseID NUMBER GENERATED ALWAYS AS IDENTITY (START WITH 300),
    courseName VARCHAR2(64),
    instructorID INT REFERENCES INSTRUCTORS(instructorID),
    maxSeats INT,
    PRIMARY KEY (courseID)
);


//enrollments table code

CREATE TABLE ENROLLMENTS (
    ticketID NUMBER GENERATED ALWAYS AS IDENTITY,
    studentID NUMBER REFERENCES STUDENTS(studentID),
    courseID NUMBER REFERENCES COURSES(courseID),
    seatNum NUMBER,
    timeStamp DATE DEFAULT SYSDATE,
    PRIMARY KEY (ticketID),
    UNIQUE (studentID, courseID),
    UNIQUE (courseID, seatNum)
);

//waitlist table code

CREATE TABLE WAITLIST (
    waitlistID NUMBER GENERATED ALWAYS AS IDENTITY,
    studentID NUMBER REFERENCES STUDENTS(studentID),
    courseID NUMBER REFERENCES COURSES(courseID),
    position NUMBER,
    PRIMARY KEY(waitlistID),
    UNIQUE (studentID, courseID),
    UNIQUE (courseID, position)

Capacity_Enforcement

CREATE OR REPLACE PROCEDURE Capacity_Enforcement(
    in_studentId IN NUMBER,
    in_courseId  IN NUMBER
) AS
    v_maxSeats  NUMBER;
    v_seatCount NUMBER;
BEGIN
    -- Lock the course row and read maxSeats
    SELECT maxSeats INTO v_maxSeats
      FROM COURSES
     WHERE courseID = in_courseId
       FOR UPDATE;

    -- Derive current fill level from ENROLLMENTS
    -- (COURSES has no seatNum column)
    SELECT COUNT(*) INTO v_seatCount
      FROM ENROLLMENTS
     WHERE courseID = in_courseId;

    IF v_seatCount < v_maxSeats THEN
        -- Seat open: enroll with next seat number
        INSERT INTO ENROLLMENTS (studentID, courseID, seatNum)
        VALUES (in_studentId, in_courseId, v_seatCount + 1);
    ELSE
        -- Course full: add to waitlist
        INSERT INTO WAITLIST (studentID, courseID, position)
        VALUES (in_studentId, in_courseId,
               (SELECT COUNT(*) + 1 FROM WAITLIST WHERE courseID = in_courseId));
    END IF;

    COMMIT;

EXCEPTION
    WHEN DUP_VAL_ON_INDEX THEN
        ROLLBACK;
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE;
END Capacity_Enforcement;

CREATE OR REPLACE PROCEDURE Enroll_Waitlisted(
    p_courseId IN NUMBER
) AS
    v_studentId NUMBER;
    v_maxSeats  NUMBER;
    v_seatCount NUMBER;
BEGIN
    -- Lock the course row and read maxSeats
    SELECT maxSeats INTO v_maxSeats
      FROM COURSES
     WHERE courseID = p_courseId
       FOR UPDATE;

    -- Count current enrollments after the drop has occurred
    SELECT COUNT(*) INTO v_seatCount
      FROM ENROLLMENTS
     WHERE courseID = p_courseId;

    IF v_seatCount < v_maxSeats THEN
        -- Find the student at Position 1
        SELECT studentID INTO v_studentId
          FROM WAITLIST
         WHERE courseID = p_courseId
           AND position = 1;

        -- Move them into ENROLLMENTS
        -- timeStamp omitted — defaults to SYSDATE automatically
        INSERT INTO ENROLLMENTS (studentID, courseID, seatNum)
        VALUES (v_studentId, p_courseId, v_seatCount + 1);

        -- Remove from waitlist
        DELETE FROM WAITLIST
         WHERE studentID = v_studentId AND courseID = p_courseId;

        -- Shift remaining positions up by 1
        UPDATE WAITLIST SET position = position - 1
         WHERE courseID = p_courseId;
    END IF;

    COMMIT;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        COMMIT; -- Waitlist empty — release FOR UPDATE lock cleanly
END Enroll_Waitlisted;
